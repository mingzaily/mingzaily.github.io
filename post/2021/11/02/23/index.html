Golang 通过通道控制携程数量

mingzaily / 2021-11-02 

func main() {
	var wg sync.WaitGroup
	var ch = make(chan bool, 10)
    for {
        wg.Add(1)
        ch <- true
        go func() {
            defer func() {
                <- ch
                wg.Done()
            }()
            // 逻辑
            ...
        }()
    }

    wg.Wait()
}
